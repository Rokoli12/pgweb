<script>
function guardarPedido() {
  var nombre = document.getElementById("nombre").value;
  var producto = document.getElementById("producto").value;
  var cantidad = document.getElementById("cantidad").value;
  
  var pedidoInfo = "Nombre: " + nombre + "\nProducto: " + producto + "\nCantidad: " + cantidad;
  
  // Creamos un objeto Blob para guardar la información
  var blob = new Blob([pedidoInfo], {type: "text/plain;charset=utf-8"});
  saveAs(blob, "pedido.txt"); // Guardamos el archivo como "pedido.txt"
}

// Función auxiliar para descargar el archivo
function saveAs(blob, filename) {
  var url = window.URL.createObjectURL(blob);
  var a = document.createElement("a");
  a.style.display = "none";
  a.href = url;
  a.download = filename;
  document.body.appendChild(a);
  a.click();
  window.URL.revokeObjectURL(url);
}
</script>
